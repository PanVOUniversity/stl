360  template<typename U>
361  std::pair<iterator, bool> insert_impl(U&& value) {
362  std::vector<NodePtr> update(MAX_LEVEL, head_);
363  NodePtr current = head_;
364 
365  for (int i = max_level_; i >= 0; --i) {
366  while (current->forward[i] && comp_(current->forward[i]->value, value)) {
367  current = current->forward[i];
368  }
369  update[i] = current;
370  }
371 
372  current = current->forward[0];
373 
374  if (current && !comp_(value, current->value) && !comp_(current->value, value)) {
375  return {iterator(current), false};
376  }
377 
378  size_type new_level = random_level();
379  if (new_level > max_level_) {
380  for (size_type i = max_level_ + 1; i <= new_level; ++i) {
381  update[i] = head_;
382  }
383  max_level_ = new_level;
384  }
385 
386  auto new_node = std::make_shared<Node>(std::forward<U>(value), new_level);
387 
388  for (size_type i = 0; i <= new_level; ++i) {
389  new_node->forward[i] = update[i]->forward[i];
390  update[i]->forward[i] = new_node;
391  }
392 
393  ++size_;
394  return {iterator(new_node), true};
395  }
